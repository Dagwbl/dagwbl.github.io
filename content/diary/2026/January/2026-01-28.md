```shell
function now {
    param(
        [switch]$r,
        [Parameter(ValueFromRemainingArguments=$true)]
        $remainingArgs
    )
    
    if ($r) {
        # Execute on remote machine
        $argsString = if ($remainingArgs) { $remainingArgs -join ' ' } else { '' }
        ssh jinpeng6@WVLIU-GD15 "powershell.exe -File ""D:\A\Jeapo's blog\assets\code\now.ps1"" $argsString"
    } else {
        # Execute locally
        if ($remainingArgs) {
            & "D:\blog\assets\code\now.ps1" @remainingArgs
        } else {
            & "D:\blog\assets\code\now.ps1"
        }
    }
```
